문제: 책 제목에 특정 keyword가 들어있는 책을 검색해서 출력하는 프로그램을 작성해보세요

```
package first;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.Scanner;

public class Test {
    public static void main(String[] args) {
        Connection con = null;
        try {
            // 1. JDBC Driver Loading
            Class.forName("oracle.jdbc.driver.OracleDriver");
            // 2. 데이터베이스 연결(Connection 생성)
            String jdbcURL = "jdbc:oracle:thin:@localhost:1521:xe";
            String id = "C##DEV";
            String pw = "1234";
            con = DriverManager.getConnection(jdbcURL, id, pw);

            String keyword = "여행";
            // SQL문장을 만들어야 해요!
            String sql = "SELECT bisbn, btitle, bauthor, bprice from book where btitle like ?";
//            sql += "FROM book WHERE btitle LIKE ?";
            PreparedStatement pstmt = con.prepareStatement(sql);
            // 아직 문장이 완성 안됐어요! => 이 상태로는 실행이 안돼요!
            pstmt.setString(1, "%" + keyword + "%");
            // 문장이 완성돼요!
            // 실행을 시켜요!
            ResultSet rs = pstmt.executeQuery();

            while(rs.next()) {
                String tmp = "";
                tmp += rs.getString("bisbn") + " ";
                tmp += rs.getString("btitle") + " ";
                tmp += rs.getString("bauthor") + " ";
                tmp += rs.getInt("bprice") + " ";
                System.out.println(tmp);
            }
            rs.close();
            pstmt.close();
            con.close();
        } catch(Exception e) {
            System.out.println("먼가 이상해요!!");
        }
    }
}
```
