package javaexam2;
import java.util.*;


public class Test04 {
    public static void main(String[] args) {
        List list = new ArrayList();

        for (int i = 0; i < 10; i++) {
            list.add((int) (Math.random()*10) + 1);
            System.out.print(list.get(i));
            System.out.print(' ');
        }
        System.out.println();

        Collections.sort(list);

        for (int j =0; j<10; j++) {
            System.out.print(list.get(j));
            System.out.print(' ');
        }
    }
}
