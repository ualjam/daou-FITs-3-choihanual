import java.util.*;

public class Test05 {
    public static void main(String[] args) {
        HashMap map = new HashMap();
        double sum = 0.0;

        for(int i = 0; i<10; i++){
            map.put(i, (int) (Math.random()*100) + 1);
        }

        for(int j = 0; j<10; j++){
            System.out.print(map.get(j));
            System.out.print(' ');
        }
        System.out.println();

        double total = 0;

        for (int k = 0; k<10; k++) {
            total += (int) map.get(k);
        }

        System.out.printf("합계: %.2f\n", total);
        System.out.printf("평균: %.2f\n", total/10);

    }
}
